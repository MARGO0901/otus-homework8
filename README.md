Allowed options:
  -h [ --help ]                  Help message
  -D [ --directory ] arg         Directory to scan
  -E [ --exclude_directory ] arg Directory to exclude
  -L [ --scan_level ] arg        Level of scan
  -s [ --min_file_size ] arg     Minimum file size
  -M [ --mask_file_name ] arg    Mask of file name
  -H [ --hash ] arg              Hash algorithm
  -S [ --block_size ] arg        Size_of_block
